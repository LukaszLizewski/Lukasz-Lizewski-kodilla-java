SELECT U.SUMMARY, P.NAME
FROM ISSUES U 
JOIN ISSUESLISTS P 
ON U.ISSUESLIST_ID = P.ID;

2)
SELECT U.SUMMARY, P.FIRSTNAME, P.LASTNAME
FROM ISSUES U
JOIN USERS P 
ON U.USER_ID_ASSIGNEDTO = P.ID;

3)
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U 
JOIN ISSUES P 
ON U.ID = P.USER_ID_ASSIGNEDTO
GROUP BY P.USER_ID_ASSIGNEDTO;
